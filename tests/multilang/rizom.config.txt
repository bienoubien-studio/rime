const renderTitle: BlocksFieldBlockRenderTitle = ({ fields, position }) => {
	const hasLink = 'link' in fields;
	const hasLabel = hasLink && 'label' in fields.link && typeof fields.link.label === 'string';
	if (!hasLabel) {
		return 'Link ' + position;
	}
	return `${fields.link.label}`;
};

const setHome: CollectionHookBeforeUpsert<PagesDoc> = async (args) => {
	const { data, api } = args;

	if (data?.home) {
		const query = {
			where: {
				home: {
					equals: true
				}
			}
		};

		const pagesIsHome = await api.collection('pages').find({ query });

		for (const page of pagesIsHome) {
			await api.collection('pages').updateById({
				id: page.id,
				data: { home: false }
			});
		}
	}

	return args;
};

const formatslug: CollectionHookBeforeUpsert<PagesDoc> = async (args) => {
	const { data, api, operation, event } = args;

	const queryPagesWithSlug = async ({ slug }: { slug: string }) => {
		const query = `where[slug][equals]=${slug}`;
		return api.collection('pages').find({
			query,
			locale: event.locals.locale
		});
	};

	if (operation === 'create' || operation === 'update') {
		if (data.slug) {
			const baseSlug = data.slug;
			let suffix = 0;
			let pagesWithCurrentSlug = await queryPagesWithSlug({ slug: baseSlug });
			while (pagesWithCurrentSlug.some((page) => page.slug === data.slug)) {
				suffix++;
				data.slug = `${baseSlug}-${suffix}`;
				pagesWithCurrentSlug = await queryPagesWithSlug({ slug: data.slug });
			}
		}

		return {
			...args,
			data
		};
	}
	return args;
};
